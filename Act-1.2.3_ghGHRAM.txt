Variable mot:chaîne de caractére
         i,j:entier
         résultat:booléen
Début
 i=1
 j=longueur(mot)
 résultat = Vrai;
    Afficher("saisir un mot")
    Saisir(mot)
       Tantque ((résultat = Vrai) et (i < j)) Faire
          Si(mot[i] = mot[j]) Alors  résultat = Vrai
                                     i = i + 1
                                     j = j - 1
                Sinon  résultat = Faux
                
                Finsi
          Finsi
       Fintantque  
Fin
